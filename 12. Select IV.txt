Agruparemos agora

SELECT carga FROM cursos                          SELECT DISTINCT carga
GROUP BY carga;                 INSTEAD OF        FROM cursos;

Os resultados são semelhantes mas, ao invés de terem sido distinguidos, foram
agrupados.

-- Quantos cursos têm a mesma carga
SELECT carga, COUNT(nome) FROM cursos
GROUP BY carga;

-- Buscar quantos cursos possuo com cada carga horária
SELECT carga, count(nome) FROM cursos
WHERE totaulas = 30
GROUP BY carga;

-- Apenas se o contador for acima de 3
SELECT carga, COUNT(nome) FROM cursos
GROUP BY carga
HAVING COUNT(nome) > 3;  -- Só trabalhamos com o campo que agrupamos

-- Exemplo complexo
SELECT ano, count(*) from cursos
WHERE totaulas > 30
GROUP BY ano
HAVING ano > '2013'
ORDER BY COUNT(*) DESC;

-- Selecionando a carga e total de horas dos cursos onde o ano seja acima de
-- 2015 agrupando por carga, onde só mostra a carga acima da média
SELECT carga, COUNT(*) FROM cursos
WHERE ano > '2015'
GROUP BY carga
HAVING carga > (SELECT AVG(carga) FROM cursos);
